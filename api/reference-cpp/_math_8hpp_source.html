134 constexpr float Math::hermite(float v1, float a1, float v2, float a2, float t) noexcept
135 {
136  float a = 2.f * (v1 - v2) + (a1 + a2);
137  float b = 3.f * (v2 - v1) - (2.f * a1) - a2;
138  float r = a;
139  r *= t;
140  r += b;
141  r *= t;
142  r += a1;
143  r *= t;
144  return r + v1;
145 }

147 constexpr float Math::catmullRom(float v1, float v2, float v3, float v4, float t) noexcept
148 {
149  float d1 = (v3 - v1) * 0.5f;
150  float d2 = (v4 - v2) * 0.5f;
151  return (2.0f * v2 - 2.0f * v3 + d1 + d2) * t * t * t + (-3.0f * v2 + 3.0f * v3 - 2.0f * d1 - d2) * t * t + d1 * t + v2;
152 }
