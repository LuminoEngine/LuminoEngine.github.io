10 class GeometryStructsHelper
11 {
12 public:
13  template<typename TRect>
14  static void clip(TRect* this_, const TRect& rect)
15  {
16  auto l = (this_->x < rect.x) ? rect.x : this_->x;
17  auto t = (this_->y < rect.y) ? rect.y : this_->y;
18  auto r = (this_->getRight() > rect.getRight()) ? rect.getRight() : this_->getRight();
19  auto b = (this_->getBottom() > rect.getBottom()) ? rect.getBottom() : this_->getBottom();
20  this_->x = l;
21  this_->y = t;
22  this_->width = r - l;
23  this_->height = b - t;
24  }
25 };
