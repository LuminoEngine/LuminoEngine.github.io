98  template<typename TChar, typename TLookuped>
99  static void SplitHelper(const TChar* begin, const TChar* end, const TChar* delim, int delimLen, StringSplitOptions option, CaseSensitivity cs, TLookuped callback)
100  {
101  if (LN_REQUIRE(begin != nullptr)) return;
102  if (LN_REQUIRE(end != nullptr)) return;
103  if (LN_REQUIRE(begin <= end)) return;
104  if (LN_REQUIRE(delim != nullptr)) return;
105 
106  delimLen = (delimLen < 0) ? static_cast<int>(strlen(delim)) : delimLen;
107  const TChar* cur = begin;
108  const TChar* tokenBegin = begin;
109 
110  while (cur < end)
111  {
112  if (*cur == *delim && (end - cur >= delimLen) && compare(cur, delim, delimLen, cs) == 0)
113  {
114  if (option == StringSplitOptions::RemoveEmptyEntries && tokenBegin == cur)
115  {
116  // 空文字列を無視する
117  }
118  else
119  {
120  callback(tokenBegin, cur);
121  }
122  cur += delimLen;
123  tokenBegin = cur;
124  }
125  else
126  {
127  cur++;
128  }
129  }
130 
131  // 最後のトークン or デリミタが1つもないとき の分
132  if (option == StringSplitOptions::RemoveEmptyEntries && tokenBegin == cur)
133  {
134  // 空文字列を無視する
135  }
136  else
137  {
138  callback(tokenBegin, cur);
139  }
140  }
