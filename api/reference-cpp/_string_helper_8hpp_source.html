100  template<typename TChar, typename TLookuped>
101  static void SplitHelper(const TChar* begin, const TChar* end, const TChar* delim, int delimLen, StringSplitOptions option, CaseSensitivity cs, TLookuped callback)
102  {
103  if (LN_REQUIRE(begin != nullptr)) return;
104  if (LN_REQUIRE(end != nullptr)) return;
105  if (LN_REQUIRE(begin <= end)) return;
106  if (LN_REQUIRE(delim != nullptr)) return;
107 
108  delimLen = (delimLen < 0) ? static_cast<int>(strlen(delim)) : delimLen;
109  const TChar* cur = begin;
110  const TChar* tokenBegin = begin;
111 
112  while (cur < end)
113  {
114  if (*cur == *delim && (end - cur >= delimLen) && compare(cur, delim, delimLen, cs) == 0)
115  {
116  if (option == StringSplitOptions::RemoveEmptyEntries && tokenBegin == cur)
117  {
118  // 空文字列を無視する
119  }
120  else
121  {
122  callback(tokenBegin, cur);
123  }
124  cur += delimLen;
125  tokenBegin = cur;
126  }
127  else
128  {
129  cur++;
130  }
131  }
132 
133  // 最後のトークン or デリミタが1つもないとき の分
134  if (option == StringSplitOptions::RemoveEmptyEntries && tokenBegin == cur)
135  {
136  // 空文字列を無視する
137  }
138  else
139  {
140  callback(tokenBegin, cur);
141  }
142  }
