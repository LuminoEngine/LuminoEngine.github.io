269 inline bool notifyException(ExceptionLevel level, const char* file, int line, const char* exprString, TArgs... args)
270 {
271  const size_t BUFFER_SIZE = 512;
272  Char str[BUFFER_SIZE] = {};
273  errorPrintf(str, BUFFER_SIZE, args...);
274 
275  TException e(str);
276  detail::Exception_setSourceLocationInfo(e, level, file, line, exprString);
277 
278  auto h = Exception::notificationHandler();
279  if (h != nullptr && h(e)) {
280  return true;
281  }
282 
283  printError(e);
284 
285  if (level == ExceptionLevel::Fatal || level == ExceptionLevel::Error) {
286  throw e;
287  }
288 
289  return true;
290 }
