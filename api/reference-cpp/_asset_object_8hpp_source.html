11 struct AssetRequiredPathSet
12 {
13  // ユーザープログラムから指定されたパス。
14  //
15  // 絶対パスまたは相対パスで、相対パスの場合は Assets フォルダからの相対パス。
16  // 初回ロードの他、リロードでも使用し、ロードのたびに AssetPath に解決して使う。
17  //
18  // load 時の処理としては、拡張子が無い場合はまず .yml であると仮定してファイルを探しに行く。
19  // - ローカルファイル・アーカイブファイルの場合は exists() するだけ
20  // - ネットワークファイルの場合は .yml で GET リクエストして、失敗するようなら候補拡張子使って繰り返す。
21  // - パフォーマンスかなり悪くなるが、「Web アプリにするなら拡張子まで明示することを推奨」でまずは行ってみる。
22  //
23  // このフィールドはシリアライズには含まない。
24  Path requiredLoadPath;
25 
26  // アセットファイル (.yml) のパス。
27  //
28  // この AssetObject 自体の serialize (save/load) で使用するファイルのパスで、
29  // load 時にファイルが存在しない場合は Empty.
30  // save するとファイルができるので、その時はここにパスが格納される。
31  //
32  // 値を持っていても、必ずファイルが存在するとは限らない。
33  // ネットワーク上のファイルの場合は、このパスで GET リクエストするまで結果はわからない。
34  // ただし、少なくとも「優先的に .yml を読み込みたい」状態であると考えることはできる。
35  //
36  // Tileset 内の Texture2D など、他のアセットの .yml に別アセットの serialize も含める場合、
37  // このフィールドは Empty となる。
38  //
39  // このフィールドはシリアライズには含まない。
40  detail::AssetPath assetFilePath;
41 
42  // この AssetObject が参照するメインのリソースファイルのパス。
43  // .png, .gltf, .wav など様々な外部ファイルで、読み取りのみで使用される。
44  //
45  // AssetObject とこれらのリソースファイルは 1:1 で対応するものとする。
46  // これは XXXX::load() でひとつのファイルを指定することと一致させるため。
47  // アセットによっては複数のリソースを扱うようなケースは考えない。
48  // （もしそれが必要なのであれば、サブクラスでカバーする）
49  //
50  // この値は、
51  // - m_requiredLoadPath が絶対パスの場合、絶対パスとなる。
52  // - m_assetFilePath が Empty の場合、
53  // - Tileset 内の Texture2D など、他アセットの .yml からの読み込みの場合は
54  // Serialize が持っているルートパスからの相対パスを表す。(★TODO: 未実装)
55  // - それ以外は Assets フォルダからの相対パスを表す。
56  // - m_assetFilePath に値がある場合、.yml ファイルからの相対パスを表す。
57  // Note: 常に m_assetFilePath 空の相対パスにした方がシンプルなのだが、
58  // 特にテクスチャや音声は .yml を使わずにロードすることがほとんどで、
59  // これらのために m_assetFilePath を構築するとかなりのメモリの無駄になる。
60  //
61  // このフィールドはシリアライズに `含める` 。
62  Path resourceFilePath;
63 
64  // resourceFilePath をベースとして、最終的に決まったリソースファイルの AssetPath。
65  //
66  // このフィールドはシリアライズには含まない。
67  detail::AssetPath finalResourceAssetFilePath;
68 };
