307  void updateValue()
308  {
309  if (Math::nearEqual(m_totalTime, 0.0f)) {
310  m_value = m_targetValue;
311  return;
312  }
313 
314  TTime time;
315  switch (m_wrapMode)
316  {
317  case AnimationWrapMode::Once:
318  time = m_currentTime;
319  break;
320  case AnimationWrapMode::Loop:
321  if (m_currentTime > m_totalTime)
322  {
323  m_currentTime = 0;
324  }
325  time = m_currentTime;
326  break;
327  case AnimationWrapMode::RoundTrip:
328  if (m_currentTime > m_totalTime)
329  {
330  m_currentTime = -m_currentTime;
331  }
332  time = std::abs(m_currentTime) / m_totalTime;
333  break;
334  }
335 
336  if (m_currentTime >= m_totalTime)
337  {
338  m_value = m_targetValue;
339  }
340  else if (m_currentTime <= 0 || m_totalTime <= 0)
341  {
342  m_value = m_startValue;
343  }
344  else if (m_easing != nullptr)
345  {
346  m_value = m_easing(m_currentTime, m_startValue, m_targetValue - m_startValue, m_totalTime);
347  }
348  else
349  {
350  m_value = m_startValue + (TValue)((m_targetValue - m_startValue) * (m_currentTime / m_totalTime)); // 線形補間
351  }
352  }
